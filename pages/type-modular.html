// gridC => grid columns
@mixin baseC($columns: 12) {
  // loop through number of columns
  @for $i from 1 through $columns {
    // add a class for child elm to .span-# a certain width
    & > .span-#{$i} {
      // calculate column span-# width
      width: spanWidth(calculateColumns($columns), $i);
    }
  }
}
